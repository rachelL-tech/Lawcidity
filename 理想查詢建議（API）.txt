A) court_units：先 upsert 審判單位（給地圖/所在地用）
INSERT INTO court_units (unit_norm, root_norm, level, county, district, address, lat, lon, updated_at)
VALUES (:unit_norm, :root_norm, :level, :county, :district, :address, :lat, :lon, now())
ON CONFLICT (unit_norm) DO UPDATE
SET
  root_norm  = EXCLUDED.root_norm,
  level      = COALESCE(EXCLUDED.level, court_units.level),
  county     = COALESCE(EXCLUDED.county, court_units.county),
  district   = COALESCE(EXCLUDED.district, court_units.district),
  address    = COALESCE(EXCLUDED.address, court_units.address),
  lat        = COALESCE(EXCLUDED.lat, court_units.lat),
  lon        = COALESCE(EXCLUDED.lon, court_units.lon),
  updated_at = now()
RETURNING id;

B) decisions：用自然鍵 upsert（placeholder → 補齊全文/URL）
你不用寫 ref_key，它會自動生成。
INSERT INTO decisions (
  court_root_norm, jyear, jcase_norm, jno,
  jid, court_unit_id, decision_date, title, full_text, pdf_url, raw, updated_at
) VALUES (
  :court_root_norm, :jyear, :jcase_norm, :jno,
  :jid, :court_unit_id, :decision_date, :title, :full_text, :pdf_url, :raw::jsonb, now()
)
ON CONFLICT (court_root_norm, jyear, jcase_norm, jno) DO UPDATE
SET
  jid           = COALESCE(decisions.jid, EXCLUDED.jid),
  court_unit_id = COALESCE(decisions.court_unit_id, EXCLUDED.court_unit_id),
  decision_date = COALESCE(decisions.decision_date, EXCLUDED.decision_date),
  title         = COALESCE(decisions.title, EXCLUDED.title),
  full_text     = COALESCE(decisions.full_text, EXCLUDED.full_text),
  pdf_url       = COALESCE(decisions.pdf_url, EXCLUDED.pdf_url),
  raw           = COALESCE(decisions.raw, EXCLUDED.raw),
  updated_at    = now()
RETURNING id, ref_key;

C) citations：插入引用邊（自動去重）
INSERT INTO citations (source_id, target_id, raw_match, match_start, match_end, snippet)
VALUES (:source_id, :target_id, :raw_match, :match_start, :match_end, :snippet)
ON CONFLICT (source_id, target_id, match_start) DO NOTHING
RETURNING id;

D) 理由段法條：去重寫入（decision_reason_statutes）
INSERT INTO decision_reason_statutes (decision_id, law, article, raw_match)
VALUES (:decision_id, :law, :article, :raw_match)
ON CONFLICT (decision_id, law, article) DO NOTHING;

INSERT INTO citation_snippet_statutes (citation_id, law, article, raw_match)

核心查詢：排行榜 / 點進去 / 地圖篩選
1) 基礎排行榜：被引用最多的判決（不加 filter）
SELECT t.ref_key, t.title, COUNT(DISTINCT c.source_id) AS citing_cases
FROM citations c
JOIN decisions t ON t.id = c.target_id
GROUP BY t.id
ORDER BY citing_cases DESC
LIMIT 50;

2) 加條件排行榜：來源判決需符合 keyword + 理由段含法條 + （可選）所在地
WITH filtered_sources AS (
  SELECT DISTINCT d.id
  FROM decisions d
  JOIN decision_reason_statutes s ON s.decision_id = d.id
  LEFT JOIN court_units cu ON cu.id = d.court_unit_id
  WHERE d.court_root_norm IN ('最高法院','臺灣高等法院','臺灣新北地方法院')  -- 你自訂
    AND d.decision_date >= (CURRENT_DATE - INTERVAL '5 years')
    AND d.full_text ILIKE '%' || :keyword || '%'
    AND s.law = :law AND s.article = :article
    -- 若要所在地就加：
    -- AND cu.county = :county
)
SELECT t.ref_key, t.title, COUNT(DISTINCT c.source_id) AS citing_cases
FROM citations c
JOIN filtered_sources fs ON fs.id = c.source_id
JOIN decisions t ON t.id = c.target_id
GROUP BY t.id
ORDER BY citing_cases DESC
LIMIT 50;

3) 點進某判決：哪些判決引用它（含 snippet）
SELECT s.ref_key AS source_ref_key, s.title AS source_title, s.decision_date,
       c.raw_match, c.snippet
FROM citations c
JOIN decisions s ON s.id = c.source_id
WHERE c.target_id = :target_id
ORDER BY s.decision_date DESC NULLS LAST
LIMIT 200;

4) 回答「這次引用在講哪條法」：對某 target 的引用片段法條做統計
SELECT css.law, css.article, COUNT(*) AS mentions
FROM citations c
JOIN citation_snippet_statutes css ON css.citation_id = c.id
WHERE c.target_id = :target_id
GROUP BY css.law, css.article
ORDER BY mentions DESC
LIMIT 30;

5) 地圖/所在地篩選（縣市/區）
SELECT d.ref_key, d.title, d.decision_date
FROM decisions d
JOIN court_units cu ON cu.id = d.court_unit_id
WHERE cu.county = :county
  AND cu.district = :district
ORDER BY d.decision_date DESC NULLS LAST
LIMIT 200;
